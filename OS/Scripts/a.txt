function check_armstrong(){
	echo " Enter the number"
	read num

t=$num
s=0
c=10
while [ $num -gt 0 ]
do
	r=$((num % c))
	i=$((r * r * r))
	s=$((s + i))
	num=$((num / c))
done

	echo $s
if [ $s == $t ]
then
	echo "ArmStrong Number"
else
	echo "Not An Armstrong number"
fi
}
